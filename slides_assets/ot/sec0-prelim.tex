% ── 概率测度空间 ─────────────────────────────────────────────────────────────
\begin{frame}{对象：Polish 空间上的概率测度 $\Pp(\X)$}
  \textbf{为什么。} 传输的两端是\emph{质量分布}，不是单个点；最自然的语言是\emph{概率测度}。

  \deflab{一句话直觉。} Polish 空间就是性质足够好的空间——可以放心地取极限、做积分、谈收敛；最常见的例子是 $\R^{n}$；完备性+可分性正是保证后续紧性与逼近论证成立的两个条件。
  \deflab{约定。} $\X,\Y$ 为\TLterm{Polish 空间}（完备可分度量空间，如 $\R^n$）。
  $\Pp(\X)$ 记其上全体\emph{概率测度}。一个 $\mu\in\Pp(\X)$ 给每个（可测）集合 $A$ 一个质量
  $\mu[A]\in[0,1]$，且 $\mu[\X]=1$。积分 $\int_{\X} \varphi \dd\mu$ 表示“按密度 $\mu$
  对检验函数 $\varphi\in C_b(\X)$ 取平均”：
  \[
    \mu \;\longmapsto\; \int_{\X} \varphi \dd\mu, \qquad \varphi\in C_b(\X).
  \]
  \begin{itemize}\small
    \item \emph{例 1（点质量）。} $\delta_{x_0}$：全部质量集中在 $x_0$，$\int\varphi\dd\delta_{x_0}=\varphi(x_0)$。
    \item \emph{例 2（经验测度）。} $\frac1n\sum_{i=1}^n\delta_{x_i}$：$n$ 个等重样本点。
    \item \emph{例 3（密度）。} $\mu=f\,\mathrm{Leb}$，$f\ge0,\ \int f=1$（如高斯）。
  \end{itemize}
  \vspace{-4pt}
  \TLtakeaway{$\Pp(\X)$ 同时容纳“离散沙堆”与“连续密度”——这正是传输需要的统一框架。}
\end{frame}

% ── 弱收敛 + 紧性 ────────────────────────────────────────────────────────────
\begin{frame}{度量“靠近”：弱收敛与 Prokhorov 紧性}
  \textbf{为什么。} 要证“最优方案\emph{存在}”，标准武器是\emph{紧性 + 下半连续}（第 1 节会用到）。
  先把“测度列收敛”说清楚。

  \medskip
  \deflab{弱（窄）收敛。} $\mu_k\wkto\mu$ 指对每个有界连续 $\varphi$，
  \[
    \int_{\X} \varphi\dd\mu_k \;\To\; \int_{\X} \varphi\dd\mu .
  \]
  直觉：沙堆\emph{整体形状}收敛，允许质量\emph{连续地移动}，但不允许质量\emph{逃逸到无穷}。

  \medskip
  \deflab{Prokhorov 定理（紧性判据）。} 一族测度\emph{胶着 (tight)} ——
  即对任意 $\eps>0$ 存在紧集 $K_\eps$ 使 $\mu[\X\setminus K_\eps]\le\eps$ 对全族成立——
  当且仅当它\emph{弱列紧}。
  \TLtakeaway{“胶着”= 质量不会偷偷跑到无穷远。它是后面所有\emph{存在性}证明的引擎。}
\end{frame}

% ── 下半连续 / 上半连续 —— 定义 ──────────────────────────────────────────────
\begin{frame}{下半连续（lsc）：代价函数的自然正则性}
  \textbf{为什么。} 证最优传输计划存在（定理 4.1），需要"紧性 + lsc $\Rightarrow$ 最小值被取到"。
  代价 $c(x,y)$ 通常不连续（如截断成本），但满足\emph{下半连续}——这已足够。

  \deflab{一句话直觉。} 下半连续就是最小化问题不会扑空的正则性——函数在跳变处取较低值，于是下确界总能被某点真正取到。
  \deflab{定义（下半连续 / lsc）。} $f:X\to\mathbb{R}\cup\{+\infty\}$
  称在 $x_0$ 处\TLterm{下半连续 (lsc)}，若
  \[
    \liminf_{x\to x_0}f(x) \;\ge\; f(x_0).
  \]
  等价刻画：$f$ 全局 lsc $\;\Longleftrightarrow\;$ 对任意 $a\in\mathbb{R}$，
  下水平集 $\{x: f(x)\le a\}$ 是\emph{闭集}。

  \deflab{上半连续 (usc)} 为对偶：$\limsup_{x\to x_0}f(x)\le f(x_0)$，
  等价于所有上水平集 $\{f\ge a\}$ 闭（即 $-f$ lsc）。

  \textbf{典型例子：} 开集 $U$ 的示性函数 $\mathbf{1}_{U}$ 是 lsc；
  闭集 $F$ 的示性函数 $\mathbf{1}_{F}$ 是 usc。
  距离函数 $d(x,\cdot)$ 是连续的（既 lsc 又 usc）。

  \vspace{-3pt}
  \TLtakeaway{下半连续 $\Longleftrightarrow$ 所有下水平集 $\{f\le a\}$ 是闭集——最小化问题的自然正则性。}
\end{frame}

      \end{tikzpicture}\par
      {\scriptsize 跳跃处取下值（实心点），\par 图形"从下方封闭"。\par 据 Villani 风格重绘。}
    \end{column}
    \begin{column}{0.59\textwidth}
      \textbf{(a) 连续函数逼近（引理 4.3 用到）}

      lsc 且有下界的 $f$ 是有界连续函数列的\emph{逐点递增极限} $f_\ell\nearrow f$。
      标准构造（inf-卷积）：
      \[
        f_\ell(x) = \inf_{y}\bigl[f(y)+\ell\,d(x,y)\bigr].
      \]

      \smallskip
      \textbf{(b) 紧集上取最小值（lsc 版 Weierstrass）}

      若 $K$ 紧，$f:K\to\mathbb{R}\cup\{+\infty\}$ lsc 且不恒 $+\infty$，
      则 $\inf_{K}f$ 被某点取到。

      \smallskip
      \emph{用途：}$\pi\mapsto\int c\,\dd\pi$ lsc，$\Pi(\mu,\nu)$ 弱紧
      $\Rightarrow$ 最优耦合存在（定理 4.1）。
    \end{column}
  \end{columns}
  \TLtakeaway{紧集上的 lsc 函数必达最小值——最优传输存在性证明的核心逻辑。}
\end{frame}

% ── 绝对连续测度 ─────────────────────────────────────────────────────────────
\begin{frame}{绝对连续测度 $\Pac$：Brenier 定理的门票}
  \textbf{为什么。} Brenier 定理（§2，$T=\nabla\varphi$）和定理 8.7（§5）均要求
  $\mu\in\Pac(\mathbb{R}^n)$——否则最优\emph{映射}可能不存在，只有耦合。

  \deflab{定义（绝对连续）。} 称 $\mu$
  \TLterm{绝对连续于} $\nu$，记 $\mu\ll\nu$，若
  $\nu[N]=0 \Rightarrow \mu[N]=0$（对所有 Borel 集 $N$）。
  在 $\mathbb{R}^n$ 上：$\mu\ll\mathrm{Leb}$ 当且仅当 $\mu$ 有密度 $f\ge0$，
  即 $\mu=f\cdot\mathrm{Leb}$，$\int f\,\dd x=1$。
  记 $\Pac(X)=\{\mu\in\Pp(X):\mu\ll\mathrm{vol}\}$。

  \medskip
  \textbf{直觉：} $\mu\in\Pac$ 意味着质量平滑散布——无原子（无 $\delta_x$），
  $\mathbb{R}^2$ 中一条线段的 $\mu$-质量为零。

  \textbf{对比：} 测度 $\mathcal{H}^{1}\!\!\restriction_{\text{线段}}$（下一帧）
  是\emph{奇异的}，$\notin\Pac(\mathbb{R}^2)$。

  \TLtakeaway{$\mu\in\Pac$ 保证最优\emph{映射}（非仅耦合）的存在；无绝对连续则 Brenier 定理失效。}
\end{frame}

% ── Hausdorff 测度（定义 + 实例） ─────────────────────────────────────────────
\begin{frame}{Hausdorff 测度 $\mathcal{H}^{k}$：$k$ 维体积与奇异测度}
  \textbf{为什么。} 例 4.9 中传输计划用 $\mathcal{H}^{1}$ 限制到线段——需知它是什么。

  \deflab{一句话直觉。} $\mathcal{H}^{k}$ 就是把长度/面积/体积推广到弯曲的、低维集合——$\mathcal{H}^{1}$ 量曲线长度，$\mathcal{H}^{2}$ 量曲面面积，从而质量可以铺在一条曲线上。
  \deflab{定义（Hausdorff $k$-测度）。} 对集合 $A\subset X$，
  \[
    \mathcal{H}^{k}(A) = \lim_{\delta\to0}\inf\!\Bigl\{
      \sum_{i}(\mathrm{diam}\,U_i)^{k} :
      A\subseteq\bigcup_{i}U_i,\;\mathrm{diam}\,U_i<\delta
    \Bigr\}
  \]
  （Carathéodory 构造，乘规范化常数后得标准度量）。

  \begin{itemize}\small
    \item $\mathcal{H}^{1}$ = 弧长；$\mathcal{H}^{2}$ = 面积；$\mathcal{H}^{n}=c_n\cdot\mathrm{Leb}^n$。
    \item 线段 $\Gamma\subset\mathbb{R}^2$：$\mathcal{H}^{1}\!\!\restriction_{\Gamma}$ 把质量分布在 $\Gamma$ 上，
      但 $\mathrm{Leb}^2[\Gamma]=0$，故 $\mathcal{H}^{1}\!\!\restriction_{\Gamma}\notin\Pac(\mathbb{R}^2)$（奇异）。
  \end{itemize}

  \TLtakeaway{$\mathcal{H}^{k}$ 把"$k$ 维体积"推广到任意度量空间；$\mathcal{H}^{1}\!\!\restriction_{\text{曲线}}$ 是奇异测度，不属于 $\Pac(\mathbb{R}^2)$。}
\end{frame}

% ── 条件测度 / disintegration ─────────────────────────────────────────────────
\begin{frame}{条件测度与 Disintegration：Gluing Lemma 的基础}
  \textbf{为什么。} 三角不等式（§3，引理 4.4）的粘合引理把
  $\pi\in\Pi(\mu,\nu)$ 和 $\sigma\in\Pi(\nu,\rho)$ 沿 $\nu$ 拼合为 $\Pi(\mu,\rho)$ 中的耦合，依赖 disintegration。

  \deflab{一句话直觉。} Disintegration 就是条件分布的测度版——$\pi_{x}$ 正是已知 $X=x$ 时 $Y$ 的分布（即概率论的 $\mathbb{P}(Y\in\cdot\mid X=x)$），整体联合分布等于对 $x$ 按 $\mu$ 平均所有条件分布。
  \deflab{定理（disintegration）。} 设 $\pi\in\Pi(\mu,\nu)$，$X$ 为 Polish 空间。
  则存在 $\mu$-a.e.\ 唯一的\TLterm{条件测度族} $\{\pi_{x}\}_{x\in X}$，使得
  \[
    \int_{X\times Y} \varphi(x,y)\,\dd\pi
    = \int_{X}\!\Bigl[\int_{Y}\varphi(x,y)\,\dd\pi_{x}(y)\Bigr]\dd\mu(x).
  \]
  简记：$\pi(\dd x\,\dd y)=\mu(\dd x)\,\pi_{x}(\dd y)$。

  \textbf{粘合构造。} 取 $\sigma(\dd y\,\dd z)=\nu(\dd y)\,\sigma_{y}(\dd z)$，定义
  $(\pi\ltimes\sigma)(\dd x\,\dd y\,\dd z)=\mu(\dd x)\,\pi_{x}(\dd y)\,\sigma_{y}(\dd z)$。
  其 $(x,z)$ 边际属于 $\Pi(\mu,\rho)$，完成粘合。

  \TLtakeaway{任意耦合可沿边际分解为条件测度族；Gluing Lemma 正是利用此分解把两段路拼成一段。}
\end{frame}



% ── 推前测度 ─────────────────────────────────────────────────────────────────
\begin{frame}{动作：用映射搬运质量 —— 推前 $\pf{T}{\mu}$}
  \textbf{为什么。} “把 $\mu$ 搬成 $\nu$”最朴素的方式是给每个 $x$ 指定一个去处 $T(x)$。

  \medskip
  \deflab{推前（pushforward）。} 给可测 $T:\X\to\Y$，定义 $\nu=\pf{T}{\mu}$ 为
  \[
    \nu[B] \;=\; \mu\!\left[T^{-1}(B)\right]\quad(\forall B),
    \qquad\Longleftrightarrow\qquad
    \int_{\Y} \varphi\dd(\pf{T}{\mu}) = \int_{\X} \varphi\!\circ\! T \dd\mu .
  \]
  这恰是\emph{质量守恒}：$T$ 把落在 $T^{-1}(B)$ 的质量全部送进 $B$。

  \medskip
  \deflab{可算的例子（1D）。} 设 $\mu=\mathrm{Unif}[0,1]$，$T(x)=x^2$。则对 $y\in(0,1)$，
  \[
    \nu[(0,y)] = \mu[\,x^2<y\,] = \mu[\,x<\sqrt y\,] = \sqrt y
    \;\Rightarrow\; \nu \text{ 的密度 } g(y)=\frac{1}{2\sqrt y}.
  \]
  \TLtakeaway{推前 = “换元公式”的测度版。记号 $\pf{T}{\mu}=\nu$ 读作“$T$ 把 $\mu$ 搬成了 $\nu$”。}
\end{frame}

% ── Monge 问题 ───────────────────────────────────────────────────────────────
\begin{frame}{问题一：Monge —— 确定性搬运（可能无解）}
  \deflab{Monge 问题 (1781)。} 给定 $\mu\in\Pp(\X),\ \nu\in\Pp(\Y)$ 与成本 $c(x,y)\ge0$，求\emph{传输映射}
  \[
    \inf_{T}\ \int_{\X} c\big(x,\,T(x)\big)\dd\mu(x)
    \subjto \pf{T}{\mu}=\nu .
  \]
  即：每粒质量\emph{整粒}送到唯一去处 $T(x)$，且送达后的分布恰为 $\nu$。

  \medskip
  \deflab{它为何会“无解”。} Monge 不允许\emph{分裂}质量。若 $\mu=\delta_0$（一个点）而
  $\nu=\tfrac12\delta_{-1}+\tfrac12\delta_{1}$（两个点），则\emph{没有任何函数} $T$ 能把一个点
  送成两个点——可行集\emph{为空}。
  \begin{itemize}\small
    \item 即便可行，目标关于 $T$ \emph{高度非凸}，约束 $\pf{T}{\mu}=\nu$ 也非线性。
  \end{itemize}
  % Manual warn-style block (avoids 0.4pt draw-border overflow from \TLwarnblock)
  \begin{tikzpicture}
    \node[draw=TLneg, fill=TLneg!8, rounded corners=3pt,
          inner sep=8pt, text width=\linewidth-17pt, align=justify] {%
      \small\textbf{\color{TLneg}陷阱}\\[0.3em]
      「最优传输映射是否存在」是\emph{很难}的问题（§9--12）。本讲义靠松弛保证\emph{总有最优方案}。
    };
  \end{tikzpicture}
\end{frame}

% ── Kantorovich 松弛 ─────────────────────────────────────────────────────────
\begin{frame}{问题二：Kantorovich —— 允许分裂的搬运（总有解）}
  \deflab{耦合 / 转移计划。} $\mu,\nu$ 的\TLterm{耦合}是 $\X\times\Y$ 上的概率测度
  $\pi$，其两个\emph{边际}分别为 $\mu,\nu$：
  \[
    \pf{(\mathrm{proj}_{\X})}{\pi}=\mu,\quad \pf{(\mathrm{proj}_{\Y})}{\pi}=\nu .
    \qquad \text{记全体耦合为 } \coup{\mu}{\nu}.
  \]
  $\pi(dx\,dy)$ = “从 $x$ 附近搬一点质量到 $y$ 附近”的量；允许一个 $x$ 的质量\emph{分裂}去多个 $y$。

  \medskip
  \deflab{Kantorovich 问题 (1942)。}
  \[
    \boxed{\ C(\mu,\nu)\;=\;\inf_{\pi\in\coup{\mu}{\nu}}\ \int_{\X\times\Y} c(x,y)\dd\pi(x,y)\ }
  \]
  \begin{itemize}\small
    \item 目标对 $\pi$ \emph{线性}，约束\emph{线性凸}，$\coup{\mu}{\nu}$ 非空（含乘积测度 $\mu\otimes\nu$）且\emph{弱紧}。
    \item $\Rightarrow$ 最优 $\pi$ \emph{总存在}（第 1 节严格证明）。Monge 是 Kantorovich 中“不分裂”的特例。
  \end{itemize}
  \TLtakeaway{把“映射”松弛成“耦合”，问题从\emph{非凸、可能无解}变为\emph{线性、总有解}——这是全部理论的起点。}
\end{frame}

% ── worked example：离散运输 ─────────────────────────────────────────────────
\begin{frame}{可算的例子：$3\times3$ 离散运输}
  设 $\mu=\tfrac13(\delta_{a_1}{+}\delta_{a_2}{+}\delta_{a_3})$，$\nu=\tfrac13(\delta_{b_1}{+}\delta_{b_2}{+}\delta_{b_3})$。
  耦合 $\pi=(\pi_{ij})$ 是\emph{双随机矩阵}的 $\tfrac13$ 倍：行和列和均为 $\tfrac13$。
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \[
        \min_{\pi\ge0}\ \sum_{i,j} c_{ij}\,\pi_{ij}
        \subjto \sum_j\pi_{ij}=\tfrac13,\ \sum_i\pi_{ij}=\tfrac13 .
      \]
      \begin{itemize}\small
        \item 这是一个\textbf{线性规划}；可行域是（缩放的）\emph{双随机矩阵}多胞形。
        \item \emph{Birkhoff 定理}：其顶点恰为\emph{置换矩阵} $\Rightarrow$ 存在\emph{最优的 Monge 映射}（一个排列）。
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \centering
      \[
        c=\begin{pmatrix} 0 & 2 & 2\\ 2 & 0 & 2\\ 2 & 2 & 0\end{pmatrix}
        \;\Rightarrow\;
        \pi^\star=\tfrac13\!\begin{pmatrix} 1&0&0\\0&1&0\\0&0&1\end{pmatrix}
      \]
      {\scriptsize 对角成本为 $0$，故恒等排列最优，$C(\mu,\nu)=0$。}
    \end{column}
  \end{columns}
  \vspace{2pt}
  \TLtakeaway{离散情形 = 线性规划 + Birkhoff：松弛（双随机）与原始（置换）在\emph{顶点}处重合。这预演了第 2 节“对偶”与第 9 章“Monge=Kantorovich”。}
\end{frame}

% ── 成本与对偶伏笔 ───────────────────────────────────────────────────────────
\begin{frame}{成本函数与对偶的“面包店”伏笔}
  \deflab{成本 $c(x,y)$。} “把一单位质量从 $x$ 送到 $y$ 的代价”。常见选择：
  \[
    c(x,y)=\abs{x-y}\ (\text{距离}),\qquad c(x,y)=\abs{x-y}^2\ (\text{二次，最重要}).
  \]
  最优总成本 $C(\mu,\nu)$ 由上页 Kantorovich 问题给出。

  \medskip
  \deflab{对偶直觉（第 2 节正式展开）。} 设想一家中介：在产地 $x$ 以价 $\psi(x)$ \emph{收货}，
  在销地 $y$ 以价 $\phi(y)$ \emph{售出}，只要价格“诚实”——
  \[
    \phi(y)-\psi(x)\le c(x,y)\quad(\forall x,y),
  \]
  其利润 $\int\phi\dd\nu-\int\psi\dd\mu$ 永不超过你自运的成本。\emph{最大利润} = \emph{最小成本}。
  \TLtakeaway{“成本视角”（搬多少）与“价格视角”（值多少）在最优处相等——这就是 Kantorovich 对偶，第 2 节的核心。}
\end{frame}

\begin{frame}{§0 小结：上路前的工具箱}
  \begin{itemize}
    \item \textbf{对象}：$\Pp(\X)$ 上的概率测度；\textbf{靠近}：弱收敛；\textbf{引擎}：Prokhorov 紧性。

  \end{itemize}

\end{frame}
